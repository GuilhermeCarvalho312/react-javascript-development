import React from "react";
import PropTypes from "prop-types";
import "./TextField.css";

/**
 * Um componente TextField que exibe uma label e uma entrada de texto.
 *
 * @param {Object} props - As propriedades do componente.
 * @param {string} props.label - A label para exibir acima da entrada de texto.
 * @param {string} props.name - O nome do input.
 * @param {string} props.value - O valor do input.
 * @param {function} props.onChange - O callback que é executado quando o valor do input é alterado.
 * @param {string} props.error - A mensagem de erro para exibir abaixo do input.
 * @returns {React.Element} Um elemento React que exibe uma label e uma entrada de texto.
 */
const TextField = ({ label, name, value, onChange, error, ...rest }) => {
  return (
    <div className="field-text">
      <label>{label}</label>
      <input name={name} value={value} onChange={onChange} {...rest} />
      {error && <p className="field-error">{error}</p>}
    </div>
  );
};

TextField.propTypes = {
  label: PropTypes.string.isRequired,
  name: PropTypes.string.isRequired,
  value: PropTypes.string.isRequired,
  onChange: PropTypes.func.isRequired,
  error: PropTypes.string,
};

export default TextField;
