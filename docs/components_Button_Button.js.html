import PropTypes from "prop-types";
import "./Button.css";

/**
 * Componente de botão reutilizável para formulários.
 *
 * @param {Object} props - As propriedades do componente.
 * @param {React.ReactNode} props.children - O conteúdo do botão.
 * @param {string} props.type - O tipo do botão (submit, reset, button).
 * @param {string} props.className - A classe CSS para o botão.
 * @param {function} props.onClick - O callback que é executado quando o botão é clicado.
 * @returns {React.Element} Um elemento React que representa o botão.
 * @example
 * <Button>Criar card</Button>
 */
const Button = ({ children, type = "button", className, onClick }) => {
  return (
    <button type={type} className={`button ${className}`} onClick={onClick}>
      {children}
    </button>
  );
};

Button.propTypes = {
  children: PropTypes.node.isRequired,
  type: PropTypes.oneOf(["submit", "reset", "button"]),
  className: PropTypes.string,
  onClick: PropTypes.func,
};

export default Button;
