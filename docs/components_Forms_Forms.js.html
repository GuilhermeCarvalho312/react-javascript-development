import React, { useState } from "react";
import "./Forms.css";
import TextField from "../TextField/TextField";
import DropDownList from "../DropDownList/DropDownList";
import Button from "../Button/Button";

/**
 * Um componente de formulário que permite ao usuário inserir dados para criar um card de colaborador.
 *
 * @returns {React.Element} Um elemento React que contém um formulário.
 */
const Forms = () => {
  const teams = ["Programação", "Frontend", "Data Science", "DevOps"];
  const [formValues, setFormValues] = useState({
    name: "",
    role: "",
    image: "",
    team: "",
  });
  const [formErrors, setFormErrors] = useState({});
  const [setIsSubmitting] = useState(false);

  const handleInputChange = (event) => {
    const { name, value } = event.target;
    setFormValues((prevState) => ({ ...prevState, [name]: value }));
  };

  const handleSubmit = (event) => {
    event.preventDefault();
    alert("Card foi criado!");
    setIsSubmitting(true);
    setFormErrors(validateFormValues(formValues));
  };

  const validateFormValues = (values) => {
    let errors = {};
    if (!values.name) {
      errors.name = "O nome é obrigatório";
    }
    if (!values.role) {
      errors.role = "O cargo é obrigatório";
    }
    if (!values.image) {
      errors.image = "A imagem é obrigatória";
    }
    if (!values.team) {
      errors.team = "O time é obrigatório";
    }
    return errors;
  };

  return (
    <section className="forms">
      <form onSubmit={handleSubmit}>
        <h2>Preencha os dados para criar o card do colaborador</h2>
        <TextField
          label="Nome"
          name="name"
          value={formValues.name}
          onChange={handleInputChange}
          error={formErrors.name}
        />
        <TextField
          label="Cargo"
          name="role"
          value={formValues.role}
          onChange={handleInputChange}
          error={formErrors.role}
        />
        <TextField
          label="Imagem"
          name="image"
          value={formValues.image}
          onChange={handleInputChange}
          error={formErrors.image}
        />
        <DropDownList
          label="Time"
          name="team"
          value={formValues.team}
          onChange={handleInputChange}
          items={teams}
          error={formErrors.team}
        />
        <Button className="form-button">Criar card</Button>
      </form>
    </section>
  );
};

export default Forms;
